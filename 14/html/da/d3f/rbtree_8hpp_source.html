38 template <typename Element, typename Compar>
39 typename RBTree<Element, Compar>::Node* RBTree<Element, Compar>::Node::setLeft(Node* lf)
40 {
41  // предупреждаем повторное присвоение
42  if (_left == lf)
43  return nullptr;
44 
45  // если новый левый — действительный элемент
46  if (lf)
47  {
48  // если у него был родитель
49  if (lf->_parent)
50  {
51  // ищем у родителя, кем был этот элемент, и вместо него ставим бублик
52  if (lf->_parent->_left == lf)
53  lf->_parent->_left = nullptr;
54  else // доп. не проверяем, что он был правым, иначе нарушение целостности
55  lf->_parent->_right = nullptr;
56  }
57 
58  // задаем нового родителя
59  lf->_parent = this;
60  }
61 
62  // если у текущего уже был один левый — отменяем его родительскую связь и вернем его
63  Node* prevLeft = _left;
64  _left = lf;
65 
66  if (prevLeft)
67  prevLeft->_parent = nullptr;
68 
69  return prevLeft;
70 }

161 template <typename Element, typename Compar>
162 typename const RBTree<Element, Compar>::Node* RBTree<Element, Compar>::find(const Element& key)
163 {
164  Node* cur = _root;
165  // проходимся по дереву, пока не доползем до нулевых листов
166  while (cur != nullptr)
167  {
168  const Element& curKey = cur->_key;
169  if (curKey == key)
170  return cur;
171 
172  if (key < curKey)
173  cur = cur->_left;
174  else // if (key > curKey)
175  cur = cur->_right;
176  }
177 
178  return nullptr;
179 }

181 template <typename Element, typename Compar >
182 typename RBTree<Element, Compar>::Node*
183  RBTree<Element, Compar>::insertNewBstEl(const Element& key)
184 {
185  Node* cur = _root;
186  Node* prev = nullptr;
187 
188  // проходимся по дереву, пока не доползем до нулевых листов
189  while (cur != nullptr)
190  {
191  prev = cur;
192  const Element& curKey = cur->getKey();
193  if (key < curKey)
194  cur = cur->_left;
195  else if (key > curKey)
196  cur = cur->_right;
197  else // запрещенный случай — дубликат
198  throw std::invalid_argument("Can't insert duplicate value");
199  }
200 
201  // время сконструировать новый нод — красный!
202  Node* newNode = new Node(key, nullptr, nullptr, nullptr, RED);
203 
204  // дошли до листочков, если пред. пустой — значит дерево пусто
205  if (prev)
206  {
207  // определим, куда вставлять относительно предыдущего: влево или вправо
208  if (key < prev->getKey())
209  prev->setLeft(newNode);
210  else
211  prev->setRight(newNode);
212  }
213  else
214  _root = newNode;
215 
216  return newNode;
217 }

220 template <typename Element, typename Compar >
221 typename RBTree<Element, Compar>::Node*
222  RBTree<Element, Compar>::rebalanceDUG(Node* nd)
223 {
224  // попадание в этот метод уже означает, что папа есть (а вот про дедушку пока не известно)
225  bool leftCase; // истина, если левый случай (дядя правый), ложь — правый (дядя левый)
226  Node* grandpa = nd->_parent->getDaddy(leftCase); // дедушка
227  if (!grandpa)
228  return nullptr; // если дедушки нет вообще, тогда все
229 
230  Node* uncle = grandpa->getChild(!leftCase); // для левого случая нужен правый дядя и наоборот.
231 
232  // если дядя такой же красный, как сам нод и его папа...
233  if (uncle && uncle->isRed())
234  {
235  uncle->setBlack(); // дядю и папу красим в черное
236  nd->_parent->setBlack();
237  grandpa->setRed(); // а дедушку — в коммунистические цвета
238 
239  // отладочное событие
240  if (_dumper)
241  _dumper->rbTreeEvent(IRBTreeDumper<Element, Compar>::DE_AFTER_RECOLOR1, this, nd);
242 
243  // теперь чередование цветов "узел-папа-дедушка-дядя" — К-Ч-К-Ч, но надо разобраться, что там
244  // с дедушкой и его предками, поэтому продолжим с дедушкой
245  return grandpa;
246  }
247 
248  // дядя черный
249  // смотрим, является ли узел правым у папочки
250  if (nd->isSpecificChildPrv(!leftCase)) // для левого случая нужен правый узел, поэтом отрицание
251  { // CASE2 в действии
252  // двигаемся в сторону папы, чтобы совершить левый переворот...
253  nd = nd->_parent;
254  if (leftCase)
255  rotLeft(nd);
256  else // ... ну, или правый
257  rotRight(nd);
258  }
259 
260  // к этому моменту nd может уже указывать не на исходный узел, а на его папу, да еще и с переворотом
261  // красим:
262  if (nd->_parent)
263  nd->_parent->setBlack(); // папу в черный
264 
265  // отладочное событие
266  if (_dumper)
267  _dumper->rbTreeEvent(IRBTreeDumper<Element, Compar>::DE_AFTER_RECOLOR3D, this, nd);
268 
269 
270  if (nd->_parent->_parent)
271  {
272  nd->_parent->_parent->setRed(); // деда в красный
273 
274  // отладочное событие
275  if (_dumper)
276  _dumper->rbTreeEvent(IRBTreeDumper<Element, Compar>::DE_AFTER_RECOLOR3G, this, nd);
277 
278  if (leftCase) // и вращаем
279  rotRight(nd->_parent->_parent);
280  else
281  rotLeft(nd->_parent->_parent);
282  }
283 
284  return nd;
285 }

288 template <typename Element, typename Compar >
289 void RBTree<Element, Compar>::rebalance(Node* nd)
290 {
291 
292  // пока папина физиономия — цвета пионерского галстука, действуем
293  while (nd && nd->isDaddyRed())
294  {
295  // локальная перебалансировка семейства "папа, дядя, дедушка" и повторная проверка
296  nd = rebalanceDUG(nd);
297  } // while (isDaddyRed())
298 
299 
300  // напоследок перекрашиваем корень в черный цвет
301  if (_root)
302  _root->setBlack();
303 
304 }

308 template <typename Element, typename Compar>
309 void RBTree<Element, Compar>::rotLeft(typename RBTree<Element, Compar>::Node* nd)
310 {
311  // правый потомок, который станет после левого поворота "выше"
312  Node* y = nd->_right;
313 
314  if (!y)
315  throw std::invalid_argument("Can't rotate left since the right child is nil");
316 
317 
318  bool xIsLeftChild; // истина, если x - левый ребенок, иначе правый
319  // имеет смысл только, если (xParent)
320  Node* xParent = nd->_parent;
321  if (xParent)
322  {
323  if (xParent->_left == nd) // левый потомок
324  xIsLeftChild = true;
325  else // иначе это может быть только правый (без проверки)
326  xIsLeftChild = false;
327  }
328 
329 
330  Node* prevYleft = y->setLeft(nd);
331  nd->setRight(prevYleft);
332 
333  // наконец, у бывшего родителя y теперь в дочках x
334  if (xParent)
335  {
336  if (xIsLeftChild)
337  xParent->setLeft(y);
338  else
339  xParent->setRight(y);
340  }
341  else // если не было родителя, значит это теперь корень
342  _root = y;
343 
344 
345  // отладочное событие
346  if (_dumper)
347  _dumper->rbTreeEvent(IRBTreeDumper<Element, Compar>::DE_AFTER_LROT, this, nd);
348 }

352 template <typename Element, typename Compar>
353 void RBTree<Element, Compar>::rotRight(typename RBTree<Element, Compar>::Node* nd)
354 {
355  // левый потомок, который станет после левого поворота "выше"
356  Node* x = nd->_left;
357 
358  if (!x)
359  throw std::invalid_argument("Can't rotate right since the left child is nil");
360 
361  bool yIsLeftChild; // истина, если y - левый ребенок, иначе правый
362  // имеет смысл только, если (yParent)
363  Node* yParent = nd->_parent;
364  if (yParent)
365  {
366  if (yParent->_left == nd) // левый потомок
367  yIsLeftChild = true;
368  else // иначе это может быть только правый (без проверки)
369  yIsLeftChild = false;
370  }
371 
372 
373  Node* prevXright = x->setRight(nd);
374  nd->setLeft(prevXright);
375 
376  // наконец, у бывшего родителя y теперь в дочках x
377  if (yParent)
378  {
379  if (yIsLeftChild)
380  yParent->setLeft(x);
381  else
382  yParent->setRight(x);
383  }
384  else // если не было родителя, значит это теперь корень
385  _root = x;
386 
387  // отладочное событие
388  if (_dumper)
389  _dumper->rbTreeEvent(IRBTreeDumper<Element, Compar>::DE_AFTER_RROT, this, nd);
390 }
