202  {
203 
204  // для любых сообщений формируем имя файла
205  char fnBuf[255];
206  sprintf(fnBuf, "%s\\dump_#_%04d.gv", _imgPath.c_str(), _imgCounter);
207 
208  // формируем информационную строку с помощью потока-строки

211

214  _evLogFile << ss.str();
215  _evLogFile << " # " << std::setw(4) << _imgCounter << std::endl;

224  const char* getInfoStr(RBTreeDumperEvent ev)
225  {
226  // храним строки прямо здесь (в единственном статическом экземпляре)
227  // данный подход не очень хорош, особенно с точки зрения потокобезопасности,
228  // однако возвращать адрес на статический объект, объявленный внутри функции,
229  // все же возможно, т.к. время жизни этого объекта дольше вызова функции.
230  // этому подходу лучше было бы предпочесть объявление этих констант статическими
231  // вне класса, что, однако, несколько громоздко из-за необходимости таскать
232  // за каждым выносням объявлением описание шаблона.
233  static const char* INFOS_ROTL = "Rot Left";

239  static const char* INFOS_ROT3G = "Recolor 3 grandpa";
240 
241 
242  // повороты
243  if (ev == IRBTreeDumper<Element, Compar>::DE_AFTER_LROT ||
244  ev == IRBTreeDumper<Element, Compar>::DE_AFTER_RROT)
245  {
246 
247  if (ev == IRBTreeDumper<Element, Compar>::DE_AFTER_LROT)
248  return INFOS_ROTL;
249  return INFOS_ROTR;
250  }
251 
252  // обычная BST-вставка

254  return INFOS_BSTINS;
255 
256

258  if (ev == IRBTreeDumper<Element, Compar>::DE_AFTER_INSERT)
259  return INFOS_INSERT;
260 
261  // NB: в принципе, в следующем if-е нет необходимости, т.к. это единственная
262  // возможная ветка, однако, если вдруг будут свдиги вверх/вниз или появятся новые
